`useCycleList` is a hook that cycles through a list of items, providing navigation functions to go to the next, previous, or a specific index.

## Usage

```tsx
import { useCycleList } from '#';
import * as React from 'react';

const colors = ['red', 'green', 'blue'];
const {
  state: currentColor,
  next,
  prev,
  go,
} = useCycleList(colors, { initialValue: 'green' });

return (
  <div>
    <p>Current color: {currentColor}</p>
    <button onClick={next}>Next</button>
    <button onClick={prev}>Prev</button>
    <button onClick={() => go(0)}>Go to first</button>
  </div>
);
```

```tsx
import { useCycleList } from '#';
import * as React from 'react';

const Component = () => {
  const names = ['Alice', 'Bob', 'Charlie'];
  const { state: currentName, index, next, prev } = useCycleList(names);

  return (
    <div>
      <p>Current Name: {currentName}</p>
      <p>Index: {index}</p>
      <button onClick={() => next(2)}>Skip Next</button>
      <button onClick={() => prev(2)}>Skip Previous</button>
    </div>
  );
};

export default Component;
```

## Type Declarations

```ts
export declare function useCycleList<T>(
  list: T[],
  options?: UseCycleListOptions<T>,
): UseCycleListReturn<T>;

export interface UseCycleListOptions<T> {
  initialValue?: T;
  fallbackIndex?: number;
  getIndexOf?: (value: T, list: T[]) => number;
}

export interface UseCycleListReturn<T> {
  state: T;
  index: number;
  next: (n?: number) => T;
  prev: (n?: number) => T;
  go: (i: number) => T;
}
```
