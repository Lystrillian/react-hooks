## Installation

```bash
npx changeelog/reactuse@latest add useToNumber
```

## Usage

```tsx
import { useToNumber } from '#';
import * as React from 'react';

const Component = () => {
  const stringValue = '42';
  const number = useToNumber(stringValue, { method: 'parseInt', radix: 10 });

  return (
    <div>
      <p>Number: {number}</p>
    </div>
  );
};

export default Component;
```

```tsx
import { useToNumber } from '#';
import * as React from 'react';

const Component = () => {
  const invalidValue = 'invalid';
  const number = useToNumber(invalidValue, { nanToZero: true });

  return (
    <div>
      <p>Number: {number}</p>
    </div>
  );
};

export default Component;
```

## Reference

### useToNumber

`useToNumber` is a hook that converts a string or a number to a numeric value using specified options.

<TypeTable

    options: {

      type: 'UseToNumberOptions',
      default: '{}',
    },
  }}
/>

### UseToNumberOptions

An interface defining optional configuration for the conversion:

<TypeTable
  type={{
    method: {
      description: 'Specifies the method to convert the value.',
      type: '"parseFloat" | "parseInt"',
      default: '"parseFloat"',
    },
    radix: {
      description: 'The radix to use when method is "parseInt".',
      type: 'number',
      default: '10',
    },
    nanToZero: {
      description:
        'If true, returns 0 instead of NaN when the conversion fails.',
      type: 'boolean',
      default: 'false',
    },
  }}
/>
