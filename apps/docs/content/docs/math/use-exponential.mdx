## Installation

```bash
npx changeelog/reactuse@latest add useExponential
```

## Usage

```tsx
import { useExponential } from '#';

const Component = () => {
  const { base, setBase, calculate } = useExponential();

  return (
    <div>
      <p>Base: {base}</p>
      <button onClick={() => setBase(Math.random())}>Set Base</button>
      <button onClick={() => console.log(calculate(2))}>Calculate</button>
    </div>
  );
};
```
