## Installation

```bash
npx changeelog/reactuse@latest add useSqrt
```

## Usage

```tsx
import { useSqrt } from '#';

const Component = () => {
  const { value, setValue, calculate } = useSqrt({ initialValue: 25 });
  const result = calculate();

  return (
    <div>
      <p>Value: {value}</p>
      <p>Square Root: {result !== null ? result : 'Error'}</p>
      <input
        type="number"
        value={value}
        onChange={(e) => setValue(Number(e.target.value))}
      />
    </div>
  );
};
```
