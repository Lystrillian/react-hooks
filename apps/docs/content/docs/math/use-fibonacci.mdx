## Installation

```bash
npx changeelog/reactuse@latest add useFibonacci
```

## Usage

```tsx
import { useFibonacci } from '#';

const Component = () => {
  const { sequence, generate } = useFibonacci();

  useEffect(() => {
    generate(10);
  }, [generate]);

  return (
    <div>
      <p>Fibonacci Sequence: {sequence.join(', ')}</p>
      <button onClick={() => generate(15)}>Generate 15 Numbers</button>
    </div>
  );
};
```
