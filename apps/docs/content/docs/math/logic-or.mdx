The `logicOr` function (also available as `or` for convenience) performs a logical OR operation on multiple provided values or functions. It efficiently checks if at least one of the provided conditions evaluates to true. The

## Usage

```tsx
import { logicOr } from '#';
import * as React from 'react';

const Component = () => {
  const value1 = true;
  const value2 = false;
  const value3 = () => false;

  const result = logicOr(value1, value2, value3);

  return (
    <div>
      <p>Logical OR result: {result.toString()}</p>
    </div>
  );
};

export default Component;
```

```tsx
import { logicOr } from '#';
import * as React from 'react';

const Component = () => {
  const [value1, setValue1] = React.useState(false);
  const [value2, setValue2] = React.useState(false);

  const result = logicOr(value1, value2, () => true);

  return (
    <div>
      <p>Logical OR result: {result.toString()}</p>
      <button onClick={() => setValue1(true)}>Set value1 to true</button>
      <button onClick={() => setValue2(true)}>Set value2 to true</button>
    </div>
  );
};

export default Component;
```
