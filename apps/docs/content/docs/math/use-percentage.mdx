## Installation

```bash
npx changeelog/reactuse@latest add usePercentage
```

## Usage

```tsx
import { usePercentage } from '#';

const Component = () => {
  const { value, setPercentage, getAbsolute } = usePercentage({
    initialValue: 25,
    total: 1000,
  });

  return (
    <div>
      <p>Percentage: {value}%</p>
      <p>Absolute Value: {getAbsolute()}</p>
      <button onClick={() => setPercentage(50)}>Set to 50%</button>
    </div>
  );
};
```
