## Installation

```bash
npx changeelog/reactuse@latest add useFactorial
```

## Usage

```tsx
import { useFactorial } from '#';

const Component = () => {
  const calculateFactorial = useFactorial();
  const result = calculateFactorial(5);

  return (
    <div>
      <p>Factorial of 5: {result !== null ? result : 'Error'}</p>
    </div>
  );
};
```
