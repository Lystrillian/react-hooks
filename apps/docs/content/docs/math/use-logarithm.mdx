## Installation

```bash
npx changeelog/reactuse@latest add useLogarithm
```

## Usage

```tsx
import { useLogarithm } from '#';

const Component = () => {
  const { base, setBase, calculate } = useLogarithm({ initialBase: 10 });
  const result = calculate(100);

  return (
    <div>
      <p>Base: {base}</p>
      <p>
        Log{base}(100) = {result !== null ? result : 'Error'}
      </p>
      <button onClick={() => setBase(2)}>Change to base 2</button>
    </div>
  );
};
```
