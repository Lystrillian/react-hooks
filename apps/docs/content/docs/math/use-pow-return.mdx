## Installation

```bash
npx changeelog/reactuse@latest add usePow
```

## Usage

```tsx
import { usePow } from '#';

const Component = () => {
  const { base, setBase, exponent, setExponent, calculate } = usePow({
    initialBase: 2,
    initialExponent: 3,
  });
  const result = calculate();

  return (
    <div>
      <p>Base: {base}</p>
      <p>Exponent: {exponent}</p>
      <p>Result: {result !== null ? result : 'Error'}</p>
      <button onClick={() => setBase(base + 1)}>Increase Base</button>
      <button onClick={() => setExponent(exponent + 1)}>
        Increase Exponent
      </button>
    </div>
  );
};
```
