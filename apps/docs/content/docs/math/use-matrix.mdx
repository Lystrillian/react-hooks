## Installation

```bash
npx changeelog/reactuse@latest add useMatrix
```

## Usage

```tsx
import { useMatrix } from '#';

const Component = () => {
  const { matrix, add, multiply, transpose } = useMatrix([
    [1, 2],
    [3, 4],
  ]);

  return (
    <div>
      <p>Matrix: {JSON.stringify(matrix)}</p>
      <button
        onClick={() =>
          add([
            [1, 1],
            [1, 1],
          ])
        }
      >
        Add
      </button>
      <button
        onClick={() =>
          multiply([
            [2, 0],
            [0, 2],
          ])
        }
      >
        Multiply
      </button>
      <button onClick={transpose}>Transpose</button>
    </div>
  );
};
```
