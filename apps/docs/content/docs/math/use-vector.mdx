## Installation

```bash
npx changeelog/reactuse@latest add useVector
```

## Usage

```tsx
import { useVector } from '#';

const Component = () => {
  const { vector, add, scale, magnitude } = useVector({ x: 3, y: 4, z: 0 });

  return (
    <div>
      <p>
        Vector: ({vector.x}, {vector.y}, {vector.z})
      </p>
      <p>Magnitude: {magnitude()}</p>
      <button onClick={() => add({ x: 1, y: 1, z: 1 })}>Add (1, 1, 1)</button>
      <button onClick={() => scale(2)}>Double</button>
    </div>
  );
};
```
