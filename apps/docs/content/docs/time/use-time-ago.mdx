## Installation

```bash
npx changeelog/reactuse@latest add useTimeAgo
```

## Usage

```tsx
import { useTimeAgo } from '#';

const Component = () => {
  const date = new Date(Date.now() - 24 * 60 * 60 * 1000); // 1 day ago
  const timeAgo = useTimeAgo(date, {
    max: 'week',
    messages: {
      justNow: 'right now',
      past: (n) => `${n} in the past`,
    },
  });

  return <div>Time ago: {timeAgo}</div>;
};
```
