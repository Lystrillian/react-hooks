## Installation

```bash
npx changeelog/reactuse@latest add useFps
```

## Usage

```tsx
import { useFps } from '#';

const Component = () => {
  const fps = useFps({
    interval: 1000,
    onFpsUpdate: (fps) => console.log(`Current FPS: ${fps}`),
  });

  return <div>Current FPS: {fps}</div>;
};
```
