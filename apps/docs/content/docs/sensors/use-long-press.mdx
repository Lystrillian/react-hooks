## Installation

```bash
npx changeelog/reactuse@latest add useLongPress
```

## Usage

```tsx
import { useLongPress } from '#';

const Component = () => {
  const longPressHandlers = useLongPress(
    () => {
      console.log('Long pressed!');
    },
    { duration: 1000 },
  );

  return (
    <div
      {...longPressHandlers}
      style={{ width: '100px', height: '100px', backgroundColor: 'skyblue' }}
    >
      Press and hold me
    </div>
  );
};
```
