## Installation

```bash
npx changeelog/reactuse@latest add useHover
```

## Usage

```tsx
import { useRef } from 'react';
import { useHover } from '#';

const Component = () => {
  const ref = useRef(null);
  const isHovering = useHover(ref, {
    onHoverChange: (hovering) => console.log('Hover state:', hovering),
  });

  return (
    <div ref={ref} style={{ background: isHovering ? 'lightblue' : 'white' }}>
      {isHovering ? 'Hovering!' : 'Hover me!'}
    </div>
  );
};
```
