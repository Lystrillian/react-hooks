`useAbs` is a custom React hook that takes a number or a function that returns a number and computes it's absolute value. The hook returns the absolute value as a number.

## Usage

```tsx
import { useAbs } from '#';
import * as React from 'react';

const Component = () => {
  const value = -10;
  const absValue = useAbs(value);

  return (
    <div>
      <p>
        Absolute value of {value}: {absValue}
      </p>
    </div>
  );
};

export default Component;
```

```tsx
import { useAbs } from '#';
import * as React from 'react';

const Component = () => {
  const [value, setValue] = React.useState(-5);

  const absValue = useAbs(() => value);

  return (
    <div>
      <p>
        Absolute value of {value}: {absValue}
      </p>
      <button onClick={() => setValue(-10)}>Update value to -10</button>
    </div>
  );
};

export default Component;
```

## Type Declarations

```typescript
export declare function useAbs(value: number | (() => number)): number;
```

### Returns

| Type     | Description                               |
