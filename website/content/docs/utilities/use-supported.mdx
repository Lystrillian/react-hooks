`useSupported` is a hook that checks if a feature or capability is supported using a callback function.

## Usage

```tsx
import { useSupported } from '#';
import * as React from 'react';

const Component = () => {
  const isLocalStorageSupported = useSupported(() => 'localStorage' in window);

  return (
    <div>
      <p>Local Storage Supported: {isLocalStorageSupported ? 'Yes' : 'No'}</p>
    </div>
  );
};

export default Component;
```

```tsx
import { useSupported } from '#';
import * as React from 'react';

const Component = () => {
  const isGeolocationSupported = useSupported(() => 'geolocation' in navigator);

  return (
    <div>
      <p>Geolocation Supported: {isGeolocationSupported ? 'Yes' : 'No'}</p>
    </div>
  );
};

export default Component;
```

## Type Declarations

```ts
export declare function useSupported(callback: SupportCheckCallback): boolean;

export declare type SupportCheckCallback = () => boolean;
```
