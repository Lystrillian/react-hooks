`useArrayDifference` calculates the difference between two arrays. It returns a new array containing elements from the first array that are not present in the second array.

## Usage

```tsx
import { useArrayDifference } from '#';

import { useState } from 'react';

const Component = () => {
  const [list1, setList1] = useState([0, 1, 2, 3, 4, 5]);
  const [list2, setList2] = useState([4, 5, 6]);

  const difference = useArrayDifference(list1, list2);

  return (
    <div>
      <p>Difference: {JSON.stringify(difference)}</p>
    </div>
  );
};

export default Component;
```

```tsx
import { useArrayDifference } from '#';
import { useState } from 'react';

const Component = () => {
  const [list1, setList1] = useState([{ id: 1 }, { id: 2 }, { id: 3 }]);
  const [list2, setList2] = useState([{ id: 2 }, { id: 4 }]);

  // Custom comparison function based on the 'id' property
  const difference = useArrayDifference(list1, list2, (a, b) => a.id === b.id);

  return (
    <div>
      <p>Difference: {JSON.stringify(difference)}</p>
    </div>
  );
};

export default Component;
```

## Type Declarations

```typescript
/**
 * A React hook that returns the difference between two arrays.
 *
 * @template T - The type of elements in the arrays.
 * @param {T[]} list - The first array.
 * @param {T[]} values - The second array.
 * @param {UseArrayDifferenceKey<T>} keyOrCompareFn - Optional.
 * A key of an object or a custom comparator function.
 * @returns {T[]} The difference between the two arrays.
 */
export function useArrayDifference<T>(
  list: T[],
  values: T[],
  keyOrCompareFn?: keyof T | UseArrayDifferenceComparatorFn<T>
): T[];

/**
 * A function that compares two elements for equality.
 *
 * @template T The type of elements being compared.
 * @param {T} element The first element to compare.
 * @param {T} value The second element to compare.
 * @returns {boolean} `true` if the elements are equal, `false` otherwise.
 */
export type UseArrayDifferenceComparatorFn<T> = (
  element: T,
  value: T
) => boolean;

/**
 * A key of an object or a custom comparator function used to determine
 * the uniqueness of elements in array.
 *
 * @template T The type of elements in the array.
 * @typedef {keyof T | UseArrayDifferenceComparatorFn<T>} UseArrayDifferenceKey
 */
export type UseArrayDifferenceKey<T> =
  | keyof T
  | UseArrayDifferenceComparatorFn<T>;
```
