A hook providing a memoized version of the `Array.findIndex` method for React, finding the index of the first element that passes the specified predicate function.

## Usage

```tsx
import { useArrayFindIndex } from '#';
import * as React from 'react';

const Component = () => {
  const [users, setUsers] = React.useState([
    { id: 1, name: 'Alice' },
    { id: 2, name: 'Bob' },
    { id: 3, name: 'Charlie' },
  ]);

  const findUserIndexById = (user: { id: number }) => user.id === 2;

  const userIndex = useArrayFindIndex(users, findUserIndexById);

  return (
    <div>
      <p>User Index: {userIndex}</p>
    </div>
  );
};

export default Component;
```

```tsx
import { useArrayFindIndex } from '#';
import * as React from 'react';

const Component = () => {
  const [books, setBooks] = React.useState([
    { title: 'Book A', author: 'Author A' },
    { title: 'Book B', author: 'Author B' },
  ]);

  const findBookByTitle = (book: { title: string }) => book.title === 'Book B';

  const bookIndex = useArrayFindIndex(books, findBookByTitle);

  return (
    <div>
      <p>Book Index: {bookIndex}</p>
    </div>
  );
};

export default Component;
```

## Type Declarations

```typescript
export declare type UseArrayFindIndexPredicate<T> = (
  element: T,
  index: number,
  array: T[]
) => boolean;

export declare function useArrayFindIndex<T>(
  list: T[],
  predicate: UseArrayFindIndexPredicate<T>
): number;
```

## Reference

| Property    | Type                            | Default                                              | Description                                                                                                      |
| ----------- | ------------------------------- | ---------------------------------------------------- | ---------------------------------------------------------------------------------------------------------------- |
| `list`      | `T[]`                           | -                                                    | The primary array to search.                                                                                     |
| `predicate` | `UseArrayFindIndexPredicate<T>` | `(element: T, index: number, array: T[]) => boolean` | A function called for every element of the `list`, returning `true` for the element to include in the new array. |

### Returns

| Type     | Description                                                                        |
| -------- | ---------------------------------------------------------------------------------- |
| `number` | The index of the first element in the array that passes the test. Otherwise, `-1`. |

## Common Pitfals

### Providing a predicate that isn't a function

Ensure thhat `predicate` is a valid function

```tsx
const books = [{ title: 'Book A' }, { title: 'Book B' }];
const invalidPredicate = 'findBookByTitle';

const bookIndex = useArrayFindIndex(books, invalidPredicate); // Error
```
