A hook providing a memoized version of the `Array.find` method for React, finding the first element that passes the specified predicate function.

## Usage

```tsx
import { useArrayFind } from '#';
import * as React from 'react';

const Component = () => {
  const [users, setUsers] = React.useState([
    { id: 1, name: 'Alice' },
    { id: 2, name: 'Bob' },
    { id: 3, name: 'Charlie' },
  ]);

  const findUserById = (user: { id: number }) => user.id === 2;

  const user = useArrayFind(users, findUserById);

  return (
    <div>
      <p>User: {user ? user.name : 'None'}</p>
    </div>
  );
};

export default Component;
```

```tsx
import { useArrayFind } from '#';
import * as React from 'react';

const Component = () => {
  const [books, setBooks] = React.useState([
    { title: 'Book A', author: 'Author A' },
    { title: 'Book B', author: 'Author B' },
    { title: 'Book C', author: 'Author A' },
  ]);

  const findBookByTitle = (book: { title: string }) => book.title === 'Book B';

  const book = useArrayFind(books, findBookByTitle);

  return (
    <div>
      <p>Book: {book ? book.title : 'None'}</p>
    </div>
  );
};

export default Component;
```

## Type Declarations

```typescript
export declare type UseArrayFindPredicate<T> = (
  element: T,
  index: number,
  array: T[]
) => boolean;

export declare function useArrayFind<T>(
  list: T[],
  predicate: UseArrayFindPredicate<T>
): T | undefined;
```

## Reference

| Property    | Type                       | Default                                              | Description                                                                                                      |
| ----------- | -------------------------- | ---------------------------------------------------- | ---------------------------------------------------------------------------------------------------------------- |
| `list`      | `T[]`                      | -                                                    | The primary array to search.                                                                                     |
| `predicate` | `UseArrayFindPredicate<T>` | `(element: T, index: number, array: T[]) => boolean` | A function called for every element of the `list`, returning `true` for the element to include in the new array. |

### Returns

| Property  | Type             | Description                                                                                              |
| --------- | ---------------- | -------------------------------------------------------------------------------------------------------- |
| `element` | `T \| undefined` | The first element in the array that satisfies the provided function. Otherwise, `undefined` is returned. |

## Common Pitfalls

### Providing a predicate that isn't a function

Ensure thhat `predicate` is a valid function

```tsx
const books = [{ title: 'Book A' }, { title: 'Book B' }];
const invalidPredicate = 'findBookByTitle';

const book = useArrayFind(books, invalidPredicate); // Error
```
