A hook providing a memoized version of the `Array.filter` method for React, filtering an array based on the specified predicate function.

## Usage

```tsx
import { useArrayFilter } from '#';
import { useState } from 'react';

const Component = () => {
  const [numbers, setNumbers] = useState([1, 2, 3, 4, 5]);
  const isEven = (number) => number % 2 === 0;

  const evenNumbers = useArrayFilter(numbers, isEven);

  return (
    <div>
      <p>Even Numbers: {evenNumbers.join(', ')}</p>
    </div>
  );
};
```

Object filtering example:

```tsx
import { useArrayFilter } from '#';
import { useState } from 'react';

const Component = () => {
  const [users, setUsers] = useState([
    { name: 'Alice', age: 25 },
    { name: 'Bob', age: 30 },
  ]);

  const isAdult = (user) => user.age >= 18;

  const adults = useArrayFilter(users, isAdult);

  return (
    <ul>
      {adults.map((user) => (
        <li key={user.name}>{user.name}</li>
      ))}
    </ul>
  );
};
```

## Type Declarations

```typescript
export type UseArrayFilterPredicate<T> = (
  element: T,
  index: number,
  array: T[]
) => boolean;

export function useArrayFilter<T>(
  list: T[],
  predicate: UseArrayFilterPredicate<T>
): T[];
```

## API Details

### Parameters

`list`: The primary array to filter.

`predicate`: A function called for every element of the `list`, returning `true` for elements to include in the new array.

### Returns

A new array containing only the elements that pass the `predicate` function.

## Common Pitfalls

### Providing a predicate that isn't a function.

```tsx
const numbers = [1, 2, 3, 4];
const invalidPredicate = 'isEven';

const evenNumbers = useArrayFilter(numbers, invalidPredicate); // Error
```
