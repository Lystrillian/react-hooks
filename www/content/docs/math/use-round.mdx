Get the average of an array reactively.

## Usage

```tsx
import { useAverage } from '#';

const list = [1, 2, 3];
const averageValue = useAverage(list); // 2
```

```tsx
import { useAverage } from '#';

const a = 1;
const b = 3;

const averageValue = useAverage(a, b); // 2
```
