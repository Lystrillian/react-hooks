The `useArrayDifference` function provides a way to determine the difference between two arrays.

## Usage

Basic Comparison

```tsx
import { useArrayDifference } from '#';
import { useState } from 'react';

const Component = () => {
  const [list1, setList1] = useState([0, 1, 2, 3, 4, 5]);
  const [list2, setList2] = useState([4, 5, 6]);

  const difference = useArrayDifference(list1, list2);

  // ... your component logic using the 'difference' state
};
```

Custom comparison logic:

```tsx
import { useArrayDifference } from 'react-use';
import { useState } from 'react';

const Component = () => {
  const [list1, setList1] = useState([{ id: 1 }, { id: 2 }, { id: 3 }]);
  const [list2, setList2] = useState([{ id: 2 }, { id: 4 }]);

  // Custom comparison function based on the 'id' property
  const difference = useArrayDifference(list1, list2, (a, b) => a.id === b.id);

  // ... your component logic
};
```

Using Key for Comparison:

```tsx
import { useArrayDifference } from 'react-use';
import { useState } from 'react';

const Component = () => {
  const [list1, setList1] = useState([{ name: 'Alice' }, { name: 'Bob' }]);
  const [list2, setList2] = useState([{ name: 'Bob' }, { name: 'Charlie' }]);

  // Compare based on the 'name' property
  const difference = useArrayDifference(list1, list2, 'name');

  // ... your component logic
};
```

## Type Declarations

```typescript
export function useArrayDifference<T>(
  list: T[],
  values: T[],
  keyOrCompareFn?: keyof T | UseArrayDifferenceComparatorFn<T>
): T[];

export type UseArrayDifferenceComparatorFn<T> = (
  element: T,
  value: T
) => boolean;
```
